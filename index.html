CREATE TABLE IF NOT EXISTS dog_photos ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, dog_id uuid REFERENCES dogs(id) ON DELETE CASCADE, user_id uuid REFERENCES auth.users(id) ON DELETE CASCADE, photo_url text NOT NULL, created_at timestamptz DEFAULT now() ); ALTER TABLE dog_photos ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "Public photo access" ON dog_photos; CREATE POLICY "Public photo access" ON dog_photos FOR SELECT USING (true); DROP POLICY IF EXISTS "Users manage own photos" ON dog_photos; CREATE POLICY "Users manage own photos" ON dog_photos FOR ALL USING (auth.uid() = user_id);
